package main

import (
    "net/http"

    "github.com/uadmin/uadmin"
)

func main() {
    // Assign RootURL value as "/admin/" and Site Name as "Login System"
    // NOTE: This code works only if database does not exist yet.
    uadmin.RootURL = "/admin/"
    uadmin.SiteName = "Login System"

    // Run the server
    uadmin.StartServer()
}
